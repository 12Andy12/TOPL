# (TOPL) Theory of programming languages and translation methods

maxStepCount = 9
wordSize = 2
abc = ["a", "b"]
N = ["S", "C", "D", "A", "B"]
rules = {
    "S": ["CD"],
    "C": ["aCA", "bCB", ""],
    "AD": ["aD"],
    "BD": ["bD"],
    "Aa": ["aA"],
    "Ab": ["bA"],
    "Ba": ["aB"],
    "Bb": ["bB"],
    "D": [""]
}

result = []
absoluteResult = set()


def normaliseResult():
    for word in result:
        is_correct = True
        for key in N:
            if key in word and key != "":
                is_correct = False
                break
        if is_correct:
            absoluteResult.add(word)


def genereate(thisWord="S", stepCount=0):
    if stepCount >= maxStepCount or thisWord == "":
        print(f"answer = {thisWord}")
        result.append(thisWord)
        return

    for key in rules:
        if key not in thisWord:
            continue
        for val in rules[key]:
            print(f"{stepCount}) {thisWord} ({key}: {val})")
            genereate(thisWord.replace(key, val, 1), stepCount + 1)


genereate()
normaliseResult()
print(result)
print(absoluteResult)
